5int8_t PaperdinkUIDateClass::fetch_data(const char *time_zone, uint8_t week_start_offset)
6{
7 struct tm timeinfo;

10 for(uint8_t i = 0; i<CONFIG_TIME_RETRY_COUNT; i++){

12
13 Serial.printf("TZ: %s\r\n", time_zone);
14 setenv("TZ", time_zone, 1);
15
16 if(!getLocalTime(&timeinfo)){
17 Serial.println("Failed to obtain time. Retrying...");
18 delay(1000);
19 if (i == CONFIG_TIME_RETRY_COUNT){
20 return -1;
21 }
22 } else {
23 break;
24 }
25 }
26
27 epoch = mktime(&timeinfo);
28
29 sscanf(ctime(&epoch), "%s %s %hhd %hhd:%hhd:%hhd %d", wday, month, &mday, &mil_hour, &min, &sec, &year);
30
31 month_num = timeinfo.tm_mon + 1;
32 // Gives offset of first day of the month with respect to Monday
33 // https://www.tondering.dk/claus/cal/chrweek.php#calcdow
34 // 0=Sunday, 1=Monday ... 6=Saturday
35 uint16_t a = (14 - month_num) / 12;
36 uint16_t y = year - a;
37 uint16_t m = month_num + (12 * a) - 2;
38
39 day_offset = (1 + y + (y / 4) - (y / 100) + (y / 400) + ((31 * m) / 12)) % 7;
40 // Change the offset based on user preference
41 day_offset = (day_offset + week_start_offset) % 7;
42
43 // Convert to 12 hour
44 if(mil_hour > 12){
45 hour = mil_hour - 12;
46 }else{
47 hour = mil_hour;
48 }
49
50 Serial.printf("Time is %d %d:%d:%d on %s on the %d/%d/%d . It is the month of %s. day_offset: %d\r\n", mil_hour, hour, min, sec, wday, mday, month_num, year, month, day_offset);
51 return 0;
52}
